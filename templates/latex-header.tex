% change URL font
\def\UrlFont{\scshape}

% document and section title color
\usepackage{xcolor}
\definecolor{titlecolor}{HTML}{DA1705}

% customize document title
\usepackage{titling}
\setlength{\droptitle}{-60pt}                 % reduce space before title
\renewcommand{\maketitlehookd}{\vspace{-2em}} % reduce space after title
\pretitle{\begin{center}\Huge\bfseries\color{titlecolor}}
\preauthor{\begin{center}\large\lineskip 0.5em\color{titlecolor}}
\postauthor{\par\end{center}}
\predate{\begin{center}\small\vspace{-0.5em}\color{titlecolor}}
\postdate{\par\end{center}}
\newcommand{\subtitle}[1]{%
  \posttitle{%
    \par\large#1\end{center}}
}

% customize section titles
\usepackage{sectsty}
\usepackage{needspace}
\sectionfont{\needspace{3\baselineskip}\color{titlecolor}}
\subsectionfont{\color{titlecolor}\mdseries\itshape}
\subsubsectionfont{\color{titlecolor}\normalfont\itshape}

% modify description lists
\usepackage{enumitem}
\setlist[description]{font=\normalfont, leftmargin=5em, style=nextline}

% just swallow any captions
\renewcommand{\caption}[1]{\relax}

% use csquotes
\usepackage[swedish=guillemets*]{csquotes}

% customize footnotes
\usepackage[marginal]{footmisc}
